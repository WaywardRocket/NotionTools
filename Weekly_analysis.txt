<script>
const HABITS_DB = "2cb390aa-2e13-83e9-8457-81016aa87e4c";
const TASKS_DB = "061390aa-2e13-8237-8af4-0138ab4474d5";
const JOURNAL_DB = "4fa390aa2e1382d6b04101eea6981632";
const ANALYSIS_DB = "67d8d6ca-90c4-4f24-9ffc-cfb0a42d0dab";

const HABITS = ["💧 Hydrate","🎵 Play Morning Playlist","🐱 Feed the Cats","🧹 Tidy Workspace & Kitchen","📖 Read Poetry","📓 Journal","📋 Set Up Rapid Log in BuJo","📖 Read Bible","🙏 Pray"];
const SPRINT_COLORS = { "1 - Urgent": "#e87f7f", "2 - Deadlines": "#e8b47f", "3 - Admin & Chores": "#e8d47f", "4 - Creative Work": "#9b7fe8" };

let token = "";
let weekOffset = 0;
let habitData = null, taskData = null, journalData = null;
let currentStart = null, currentEnd = null;

function setStep(n) {
  [1,2,3,4,5].forEach(i => {
    const el = document.getElementById("step"+i);
    el.className = "step" + (i < n ? " done" : i === n ? " active" : "");
  });
}

function showStatus(msg, type) {
  const el = document.getElementById("status-msg");
  el.textContent = msg;
  el.className = "status " + type;
}

function saveToken() {
  const val = document.getElementById("token-input").value.trim();
  if (!val) {
    document.getElementById("setup-error").innerHTML = '<div class="status error" style="margin-top:10px;">Please paste your integration token.</div>';
    return;
  }
  token = val;
  document.getElementById("setup-section").style.display = "none";
  document.getElementById("main-section").style.display = "block";
  loadWeek();
}

function getWeekRange(offset) {
  const now = new Date();
  const monday = new Date(now);
  monday.setDate(now.getDate() - ((now.getDay() + 6) % 7) + offset * 7);
  monday.setHours(0,0,0,0);
  const sunday = new Date(monday);
  sunday.setDate(monday.getDate() + 6);
  return { start: monday, end: sunday };
}

function fmtDate(d) { return d.toISOString().split("T")[0]; }
function fmtDisplay(d) { return d.toLocaleDateString("en-US", { month: "short", day: "numeric", year: "numeric" }); }
function changeWeek(dir) { weekOffset += dir; loadWeek(); }

async function callClaude(prompt, tools) {
  const body = { model: "claude-sonnet-4-20250514", max_tokens: 1500, messages: [{ role: "user", content: prompt }] };
  if (tools) body.tools = tools;
  const res = await fetch("https://api.anthropic.com/v1/messages", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(body)
  });
  const data = await res.json();
  if (data.error) throw new Error(data.error.message);
  return data;
}

async function loadWeek() {
  const { start, end } = getWeekRange(weekOffset);
  currentStart = start; currentEnd = end;
  document.getElementById("week-label").textContent = fmtDisplay(start) + " – " + fmtDisplay(end);
  document.getElementById("loading-state").style.display = "block";
  document.getElementById("data-section").style.display = "none";
  document.getElementById("error-state").style.display = "none";
  document.getElementById("status-msg").className = "status";
  setStep(1);

  try {
    // Fetch habits
    const habitPrompt = `Fetch Good Habits data from Notion for ${fmtDate(start)} to ${fmtDate(end)}.
POST https://api.notion.com/v1/databases/${HABITS_DB}/query
Authorization: Bearer ${token}
Notion-Version: 2022-06-28
Body: {"filter":{"and":[{"property":"Date","date":{"on_or_after":"${fmtDate(start)}"}},{"property":"Date","date":{"on_or_before":"${fmtDate(end)}"}}]}}

Count how many entries exist per habit name and how many have Habit Done = true.
Return ONLY JSON: {"days":<unique dates count>,"habits":[{"name":<name>,"done":<done count>,"total":<total entries>}]}`;

    const habitRes = await callClaude(habitPrompt, [{type:"web_search_20250305",name:"web_search"}]);
    const habitText = habitRes.content?.find(b => b.type === "text")?.text || "{}";
    habitData = JSON.parse(habitText.replace(/```json|```/g,"").trim());
    setStep(2);

    // Fetch tasks
    const taskPrompt = `Fetch Tasks data from Notion for ${fmtDate(start)} to ${fmtDate(end)}.
POST https://api.notion.com/v1/databases/${TASKS_DB}/query
Authorization: Bearer ${token}
Notion-Version: 2022-06-28
Body: {"filter":{"and":[{"property":"Due","date":{"on_or_after":"${fmtDate(start)}"}},{"property":"Due","date":{"on_or_before":"${fmtDate(end)}"}}]}}

Count tasks by Status (Done, Missed, TO DO, DOING) and by Sprint category.
Also calculate average Time Variance where available.
Return ONLY JSON: {"total":<n>,"done":<n>,"missed":<n>,"in_progress":<n>,"sprints":{"1 - Urgent":<n>,"2 - Deadlines":<n>,"3 - Admin & Chores":<n>,"4 - Creative Work":<n>},"avg_time_variance":<number or null>,"life_areas":[<list of unique life area names that had tasks>]}`;

    const taskRes = await callClaude(taskPrompt, [{type:"web_search_20250305",name:"web_search"}]);
    const taskText = taskRes.content?.find(b => b.type === "text")?.text || "{}";
    taskData = JSON.parse(taskText.replace(/```json|```/g,"").trim());
    setStep(3);

    // Fetch journal mood/energy
    const journalPrompt = `Fetch Daily Journal entries from Notion for ${fmtDate(start)} to ${fmtDate(end)}.
POST https://api.notion.com/v1/databases/${JOURNAL_DB}/query
Authorization: Bearer ${token}
Notion-Version: 2022-06-28
Body: {"filter":{"and":[{"property":"Date","date":{"on_or_after":"${fmtDate(start)}"}},{"property":"Date","date":{"on_or_before":"${fmtDate(end)}"}}]}}

List all unique Mood and Energy values that appear across entries.
Return ONLY JSON: {"entries":<count>,"moods":[<list of mood strings>],"energies":[<list of energy strings>]}`;

    const journalRes = await callClaude(journalPrompt, [{type:"web_search_20250305",name:"web_search"}]);
    const journalText = journalRes.content?.find(b => b.type === "text")?.text || "{}";
    journalData = JSON.parse(journalText.replace(/```json|```/g,"").trim());

    document.getElementById("loading-state").style.display = "none";
    document.getElementById("data-section").style.display = "block";
    renderData();
  } catch(e) {
    document.getElementById("loading-state").style.display = "none";
    document.getElementById("error-state").style.display = "block";
    document.getElementById("error-state").innerHTML = `<div class="status error">Could not load data: ${e.message}</div>`;
  }
}

function renderData() {
  const days = habitData?.days || 0;
  const habits = habitData?.habits || [];
  const totalHabitSlots = habits.reduce((s,h) => s + (h.total||0), 0);
  const totalHabitDone = habits.reduce((s,h) => s + (h.done||0), 0);
  const habitPct = totalHabitSlots > 0 ? Math.round((totalHabitDone/totalHabitSlots)*100) : 0;
  const taskDone = taskData?.done || 0;
  const taskMissed = taskData?.missed || 0;
  const taskTotal = taskData?.total || 0;
  const completionRate = taskTotal > 0 ? Math.round((taskDone/taskTotal)*100) : 0;
  const variance = taskData?.avg_time_variance;

  const hClass = habitPct >= 70 ? "good" : habitPct >= 40 ? "mid" : "low";
  const tClass = completionRate >= 70 ? "good" : completionRate >= 40 ? "mid" : "low";

  document.getElementById("metrics-row").innerHTML = `
    <div class="metric"><div class="metric-label">Days logged</div><div class="metric-val purple">${days}</div></div>
    <div class="metric"><div class="metric-label">Habit completion</div><div class="metric-val ${hClass}">${habitPct}%</div></div>
    <div class="metric"><div class="metric-label">Tasks done</div><div class="metric-val ${tClass}">${taskDone}/${taskTotal}</div></div>
    <div class="metric"><div class="metric-label">Tasks missed</div><div class="metric-val ${taskMissed > 3 ? 'low':'mid'}">${taskMissed}</div></div>
    ${variance !== null && variance !== undefined ? `<div class="metric"><div class="metric-label">Avg time variance</div><div class="metric-val ${Math.abs(variance) < 15 ? 'good':'mid'}">${variance > 0 ? '+':''}${Math.round(variance)}m</div></div>` : ''}
  `;

  const sortedHabits = [...habits].sort((a,b) => (b.done/Math.max(b.total,1)) - (a.done/Math.max(a.total,1)));
  document.getElementById("habits-list").innerHTML = sortedHabits.map(h => {
    const pct = h.total > 0 ? Math.round((h.done/h.total)*100) : 0;
    const bc = pct >= 70 ? "var(--green)" : pct >= 40 ? "var(--yellow)" : "var(--red)";
    return `<div class="habit-item">
      <span class="habit-name">${h.name}</span>
      <div class="habit-bar-wrap"><div class="habit-bar" style="width:${pct}%;background:${bc};"></div></div>
      <span class="habit-count">${h.done}/${h.total}</span>
    </div>`;
  }).join("");

  const sprints = taskData?.sprints || {};
  document.getElementById("sprint-list").innerHTML = Object.entries(sprints).map(([name, count]) => `
    <div class="sprint-item">
      <div class="sprint-dot" style="background:${SPRINT_COLORS[name]||'var(--purple)'};"></div>
      <span class="sprint-name">${name}</span>
      <span class="sprint-count">${count} tasks</span>
    </div>
  `).join("");

  generateAnalysis();
}

async function generateAnalysis() {
  setStep(4);
  const el = document.getElementById("analysis-text");
  el.textContent = "Generating your weekly analysis...";
  el.className = "analysis-text loading pulsing";
  document.getElementById("regen-btn").disabled = true;

  const weekStr = fmtDisplay(currentStart) + " – " + fmtDisplay(currentEnd);
  const habits = habitData?.habits || [];
  const topHabits = habits.filter(h => h.total > 0 && (h.done/h.total) >= 0.7).map(h => h.name).join(", ") || "none";
  const lowHabits = habits.filter(h => h.total > 0 && (h.done/h.total) < 0.4).map(h => h.name).join(", ") || "none";
  const sprints = taskData?.sprints || {};
  const topSprint = Object.entries(sprints).sort((a,b) => b[1]-a[1])[0]?.[0] || "unknown";
  const lifeAreas = taskData?.life_areas || [];
  const moods = journalData?.moods || [];
  const energies = journalData?.energies || [];
  const variance = taskData?.avg_time_variance;

  try {
    const res = await callClaude(`You are a warm, insightful life coach for a neurodivergent (AuDHD) freelance illustrator and designer who is building healthy habits and a creative practice.

Here is their week of ${weekStr}:

MORNING HABITS:
- Overall completion: ${Math.round(((habitData?.habits||[]).reduce((s,h)=>s+(h.done||0),0)/Math.max((habitData?.habits||[]).reduce((s,h)=>s+(h.total||0),0),1))*100)}%
- Strong habits (≥70%): ${topHabits}
- Needs attention (<40%): ${lowHabits}

TASKS:
- Total: ${taskData?.total||0}, Done: ${taskData?.done||0}, Missed: ${taskData?.missed||0}
- Task completion rate: ${taskData?.total > 0 ? Math.round(((taskData?.done||0)/(taskData?.total||1))*100) : 0}%
- Most active sprint: ${topSprint}
- Sprint breakdown: ${JSON.stringify(sprints)}
- Life areas covered: ${lifeAreas.join(", ") || "not tracked yet"}
- Average time variance: ${variance !== null && variance !== undefined ? (variance > 0 ? "+" + Math.round(variance) + " min (taking longer than estimated)" : Math.round(variance) + " min (finishing faster than estimated)") : "not enough data yet"}

MOOD & ENERGY (from journal):
- Moods this week: ${moods.join(", ") || "not logged"}
- Energy levels: ${energies.join(", ") || "not logged"}

Write a personal weekly analysis in 4 short paragraphs:
1. Honest overall read of the week — what the data says about how it actually went, not how it felt
2. Wins — specific habits and tasks done well, patterns worth celebrating
3. Growth edges — what needs attention next week, with one tiny concrete suggestion per area
4. Cross-pattern insight — something interesting about how mood/energy/habits/tasks relate to each other this week, or a trend worth watching

Tone: warm, direct, curious. Like a smart friend reviewing your week with you. No bullet points. No headers. Flowing paragraphs. Under 250 words.`);

    const text = res.content?.find(b => b.type === "text")?.text || "Could not generate analysis.";
    el.textContent = text;
    el.className = "analysis-text";
    document.getElementById("regen-btn").disabled = false;
    await saveToNotion(text);
  } catch(e) {
    el.textContent = "Could not generate analysis. Try again.";
    el.className = "analysis-text";
    document.getElementById("regen-btn").disabled = false;
  }
}

async function saveToNotion(analysis) {
  setStep(5);
  showStatus("Saving to Notion...", "success");

  const weekLabel = fmtDisplay(currentStart) + " – " + fmtDisplay(currentEnd);
  const habits = habitData?.habits || [];
  const habitPct = Math.round((habits.reduce((s,h)=>s+(h.done||0),0)/Math.max(habits.reduce((s,h)=>s+(h.total||0),0),1))*100);
  const sprints = taskData?.sprints || {};
  const topSprint = Object.entries(sprints).sort((a,b)=>b[1]-a[1])[0]?.[0] || "";
  const lowHabits = habits.filter(h=>h.total>0&&(h.done/h.total)<0.4).map(h=>h.name).join(", ")||"none";
  const moods = (journalData?.moods||[]).join(", ")||"";
  const energies = (journalData?.energies||[]).join(", ")||"";

  const prompt = `Save a weekly analysis entry to Notion.
POST https://api.notion.com/v1/pages
Authorization: Bearer ${token}
Notion-Version: 2022-06-28
Content-Type: application/json
Body:
{
  "parent": {"database_id": "${ANALYSIS_DB}"},
  "properties": {
    "Week": {"title": [{"text": {"content": "${weekLabel}"}}]},
    "Week Start": {"date": {"start": "${fmtDate(currentStart)}"}},
    "Habit Completion %": {"number": ${habitPct}},
    "Tasks Completed": {"number": ${taskData?.done||0}},
    "Tasks Missed": {"number": ${taskData?.missed||0}},
    "Top Sprints": {"rich_text": [{"text": {"content": "${topSprint.replace(/"/g,'\\"')}"}}]},
    "Neglected Life Areas": {"rich_text": [{"text": {"content": "${lowHabits.replace(/"/g,'\\"')}"}}]},
    "Avg Mood": {"rich_text": [{"text": {"content": "${moods.replace(/"/g,'\\"')}"}}]},
    "Avg Energy": {"rich_text": [{"text": {"content": "${energies.replace(/"/g,'\\"')}"}}]},
    "AI Analysis": {"rich_text": [{"text": {"content": ${JSON.stringify(analysis)}}}]}
  }
}
Return only: {"saved":true} or {"saved":false,"error":"..."}`;

  try {
    const res = await callClaude(prompt, [{type:"web_search_20250305",name:"web_search"}]);
    const text = res.content?.find(b=>b.type==="text")?.text||"";
    if (text.includes('"saved":true') || text.includes('"saved": true')) {
      showStatus("✓ Analysis saved to your Weekly Analysis database in Notion.", "success");
    } else {
      showStatus("Analysis generated but could not save automatically. Use Copy to save manually.", "error");
    }
  } catch(e) {
    showStatus("Analysis generated but could not save: " + e.message, "error");
  }
}
